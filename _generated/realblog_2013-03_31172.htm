// 二維向量
type Vector2D(dx:float, dy:float) =     // 向量預先計算的長度    
let length = sqrt(dx*dx + dy*dy)    /// 在 X 軸上的位移    
member v.DX = dx    /// 在 Y 軸上的位移    
member v.DY = dy    /// 向量的長度    
member v.Length = length    // 依照一個常數重新調整向量    
member v.Scale(k) = Vector2D(k*dx, k*dy)
